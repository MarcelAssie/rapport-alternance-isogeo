% Cadre temporaire affiché lorsqu'une image n'existe pas.
\newcommand{\FramedPlaceholder}[3]{%
    \fbox{%
        \begin{minipage}[c][#2][c]{%
            \dimexpr#1-2\fboxsep-2\fboxrule\relax
        }
            \centering
            \footnotesize
            \color{ReportGray}
            #3
        \end{minipage}%
    }%
}

% Affiche une image lorsqu'elle existe.
% Sinon, affiche automatiquement un cadre temporaire.
\newcommand{\ReportImageOrPlaceholder}[4]{%
    \IfFileExists{figures/#1}{%
        \includegraphics[
            width=#2,
            height=#3,
            keepaspectratio
        ]{figures/#1}%
    }{%
        \FramedPlaceholder{#2}{#3}{#4}%
    }%
}

        {Le fichier #1.tex est introuvable}%
    }%
}
